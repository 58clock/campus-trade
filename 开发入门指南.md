```java
private final ProductMapper productMapper;
private final UserMapper userMapper;

public Result<PageResult<ProductVO>> listProducts(ProductListRequest query, int page, int size) {
    LambdaQueryWrapper<Product> wrapper = new LambdaQueryWrapper<>();
    wrapper.eq(Product::getStatus, "ON_SALE");  // 只查在售商品

    // 关键词搜索（标题 + 描述）
    if (query.getKeyword() != null && !query.getKeyword().isBlank()) {
        wrapper.and(w -> w.like(Product::getTitle, query.getKeyword())
                         .or()
                         .like(Product::getDescription, query.getKeyword()));
    }
    // 分类筛选
    if (query.getCategory() != null && !query.getCategory().isBlank()) {
        wrapper.eq(Product::getCategory, query.getCategory());
    }
    // 价格区间
    if (query.getMinPrice() != null) {
        wrapper.ge(Product::getPrice, query.getMinPrice());
    }
    if (query.getMaxPrice() != null) {
        wrapper.le(Product::getPrice, query.getMaxPrice());
    }
    // 排序
    if ("price_asc".equals(query.getSort())) {
        wrapper.orderByAsc(Product::getPrice);
    } else if ("price_desc".equals(query.getSort())) {
        wrapper.orderByDesc(Product::getPrice);
    } else {
        wrapper.orderByDesc(Product::getCreatedAt);  // 默认最新
    }

    Page<Product> page = new Page<>(page, size);
    Page<Product> result = productMapper.selectPage(page, wrapper);

    // 转换成 VO 返回
    List<ProductVO> vos = result.getRecords().stream().map(p -> {
        ProductVO vo = new ProductVO();
        vo.setId(p.getId());
        vo.setTitle(p.getTitle());
        vo.setDescription(p.getDescription());
        vo.setPrice(p.getPrice());
        vo.setOriginalPrice(p.getOriginalPrice());
        vo.setCategory(p.getCategory());
        vo.setConditionLevel(p.getConditionLevel());
        vo.setViewCount(p.getViewCount());
        vo.setCreatedAt(p.getCreatedAt());
        // 查卖家昵称
        User seller = userMapper.selectById(p.getUserId());
        if (seller != null) vo.setSellerName(seller.getNickname());
        return vo;
    }).collect(Collectors.toList());

    return Result.ok(new PageResult<>(vos, result.getTotal(), page, size));
}
```

**注意：** `ProductService` 类上要加 `@RequiredArgsConstructor`，把 `productMapper` 和 `userMapper` 声明为 `private final` 字段。

        for (MultipartFile img : images) {
            if (!img.isEmpty()) {
                String filename = System.currentTimeMillis() + "_" + img.getOriginalFilename();

    }

    productMapper.insert(p);

    ProductVO vo = new ProductVO();
    vo.setId(p.getId());
    vo.setTitle(p.getTitle());

## 三、成员 C — 订单 + 留言模块

你的后端 OrderService 8 个方法 + MessageService 4 个方法全是空壳。**最关键是订单状态机不能出错。**

### 第1步：理解订单状态机

```
买家点击"我想要" → PENDING_PAYMENT（待付款）
       ↓ 买家付款
   PENDING_SHIPMENT（待发货）
       ↓ 卖家发货
   PENDING_RECEIPT（待收货）
       ↓ 买家确认收货
   COMPLETED（已完成）

   任意 PENDING_PAYMENT 状态 → CANCELLED（已取消）
```

每个状态转换都要校验：
- **当前状态对不对**（比如不能从 COMPLETED 跳到 PENDING_SHIPMENT）
- **操作人身份对不对**（买家不能点"发货"，卖家不能点"收货"）

### 第3步：实现 createOrder

```java
public Result<OrderVO> createOrder(Long buyerId, Long productId) {
    Product product = productMapper.selectById(productId);
    if (product == null) return Result.fail("商品不存在");
    if (!"ON_SALE".equals(product.getStatus())) return Result.fail("商品已下架或已售出");
    if (product.getUserId().equals(buyerId)) return Result.fail("不能购买自己的商品");

    // 检查是否有未完成的订单
    Long existCount = orderMapper.selectCount(
        new LambdaQueryWrapper<Order>()
            .eq(Order::getBuyerId, buyerId)
            .eq(Order::getProductId, productId)
            .notIn(Order::getStatus, "COMPLETED", "CANCELLED"));
    if (existCount > 0) return Result.fail("你已有该商品的未完成订单");

    Order order = new Order();
    order.setBuyerId(buyerId);
    order.setSellerId(product.getUserId());
    order.setProductId(productId);
    order.setAmount(product.getPrice());
    order.setStatus(OrderStatus.PENDING_PAYMENT.name());
    orderMapper.insert(order);

    // 锁定商品
    product.setStatus("LOCKED");
    productMapper.updateById(product);

    // 返回订单 VO
    return getOrderVO(order);
}
```

    if (!order.getStatus().equals(expected.name())) {

    }
    if (isBuyer && !order.getBuyerId().equals(userId)) {
        return Result.fail("只有买家才能操作");

    if (!isBuyer && !order.getSellerId().equals(userId)) {

    }
}

@Command(name = "list-products", description = "查看商品列表")
public class ListProductsCommand implements Runnable {
    @Option(names = {"-k", "--keyword"}) String keyword;
    @Option(names = {"-c", "--category"}) String category;
    @Option(names = {"-p", "--page"}) int page = 1;

    public void run() {
        // 用 java.net.http.HttpClient 调 GET /api/products
        String url = "http://localhost:8081/api/products?page=" + page;
        if (keyword != null) url += "&keyword=" + URLEncoder.encode(keyword, StandardCharsets.UTF_8);
        if (category != null) url += "&category=" + category;
        // ... 发送 HTTP 请求，格式化输出
    }
}

### 订单状态校验模板

private void checkStatus(Order order, OrderStatus expected, boolean isBuyer, Long userId) {
    if (!order.getStatus().equals(expected.name()))
        throw new BusinessException("订单状态不正确");
    if (isBuyer && !order.getBuyerId().equals(userId))
        throw new BusinessException("只有买家才能操作");
    if (!isBuyer && !order.getSellerId().equals(userId))
        throw new BusinessException("只有卖家才能操作");
}
